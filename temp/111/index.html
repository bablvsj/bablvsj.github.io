<template>

    <canvas id="wave3"></canvas>

<script>
export default {
  name: "Canvas",
  props: {
    bgColor: {
      default: "none",
    }, // èƒŒæ™¯è‰²
    waveAllHeight: { default: 160 }, // æ³¢æµªçš„æ•´ä½“é«˜åº¦

          "#f3f3f3",
          "#f3f3f3",
          "#f3f3f3",
        ];
      },
    },
    // æ³¢æµªé€Ÿç‡
    waveTime: {
      default () {
        return [4000, 4000, 4000];
      },
    },
  },

    return {};
  },
  mounted () {
    this.wavePlay("wave1", 140, this.waveColor[0], this.waveTime[0]);
    this.wavePlay("wave2", 140, this.waveColor[1], this.waveTime[1]);
    this.wavePlay("wave3", 140, this.waveColor[2], this.waveTime[2]);
  },
  methods: {
    wavePlay ($canvasID, $progress, $maveColor, $time) {
      const that = this;
      let waveWidth = 3300, // æ³¢æµªé•¿åº¦
        offset = 0,
        waveHeight = that.waveHeight, // æ³¢æµªèµ·ä¼é«˜åº¦
        waveCount = that.waveCount, // æ³¢æµªä¸ªæ•°
        startX = -1200,
        startY = 212, // canvas é«˜åº¦
        progress = $progress, // æ³¢æµªä½ç½®é«˜åº¦
        d2 = waveWidth / waveCount, // å•ä¸ªæ³¢æµªçš„å®½åº¦
        d = d2 / 2,
        hd = d / 2,
        c = document.getElementById($canvasID),
        ctx = c.getContext("2d");
      c.width = 1920; // ç”»å¸ƒå®½åº¦
      c.height = that.waveAllHeight; // ç”»å¸ƒé«˜åº¦
      function move () {
        offset -= 5;
        if (-1 * offset === d2) {
          offset = 0;
        }
        ctx.clearRect(0, 0, c.width, c.height);
        ctx.fillStyle = $maveColor; // ç”»å¸ƒå¡«å……è‰²
        ctx.beginPath();
        let offsetY = startY - progress;
        // ç»˜åˆ¶è´å¡å°”æ›²çº¿
        ctx.moveTo(startX - offset, offsetY); // å¼€å§‹ç‚¹
        for (let i = 0; i < waveCount; i++) {
          let dx = i * d2;
          let offsetX = dx + startX - offset;
          ctx.quadraticCurveTo(
            offsetX + hd,
            offsetY + waveHeight,
            offsetX + d,
            offsetY
          );
          ctx.quadraticCurveTo(
            offsetX + hd + d,
            offsetY - waveHeight,
            offsetX + d2,
            offsetY
          );
        }
        ctx.lineTo(startX + waveWidth, 3000);
        ctx.lineTo(startX, 0);
        ctx.fill();
        setTimeout(move, $time / 60); // é€Ÿåº¦
      }

      move();
    },
  },
};
</script>
