methods:{
	brightKeyword(val) {
		let keyword = this.searchIdCard.toString();
		if (val.indexOf(keyword) !== -1) {
			return val.replace(keyword, `<span style="color:#3995FF;font-weight:bold;">${keyword}</span>`)
		} else {
					return val
		}
	},
	handInputSearch(e){
		clearTimeout(this.searchTime);

			this.searchForm.industry_code = e.detail.value;
			this.searchForm.parent_indu = "";
			this.getDataist(this.searchForm);
		}, 500)
    },
    getDataist(){
        Service.apiIndustryList(fd).then(res => {
