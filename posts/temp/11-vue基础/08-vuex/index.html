let store = new Vuex.Store({
  state: {
    tableHeight:0,
    tableSHeight:0,
    tableTHeight:0,
  },
  getters: {
    getCommonHeight: (state) => (typeKey) => {
      return state.tableHeight + "px"
    },
  },
  mutations: {
    changeTableHeight(state,payload){
      state.tableHeight = payload.height;
      state.tableSHeight = payload.secondH;
      state.tableTHeight = payload.thirdH;
    }
  },
  actions: {
  	changeTableHeightSetTime(ctx){
       setTimeout(()=>{
           ctx.commit('changeTableHeight', {height:100,secondH:200,thirdH:300});
       },2000)  
    },
  }
})

export default store
